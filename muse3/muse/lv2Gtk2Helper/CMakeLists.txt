## install either user-compiled or precompiled lv2 gtk2 helpers in module dir

IF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/build/lv2Gtk2Helper32.so")

  install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/build/lv2Gtk2Helper32.so
          DESTINATION ${MusE_MODULES_DIR}
          PERMISSIONS WORLD_READ WORLD_EXECUTE OWNER_READ OWNER_EXECUTE OWNER_WRITE GROUP_READ GROUP_EXECUTE)
       
ELSE(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/build/lv2Gtk2Helper32.so")

  install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/lv2Gtk2Helper32.so
          DESTINATION ${MusE_MODULES_DIR}
          PERMISSIONS WORLD_READ WORLD_EXECUTE OWNER_READ OWNER_EXECUTE OWNER_WRITE GROUP_READ GROUP_EXECUTE)
          
ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/build/lv2Gtk2Helper32.so")
